130static scpi_bool_t processCommand(scpi_t * context) {
131 const scpi_command_t * cmd = context->param_list.cmd;
132 lex_state_t * state = &context->param_list.lex_state;
133 scpi_bool_t result = TRUE;
134
135 /* conditionaly write ; */
136 writeSemicolon(context);
137
138 context->cmd_error = FALSE;
139 context->output_count = 0;
140 context->input_count = 0;
141 context->arbitrary_reminding = 0;
142
143 /* if callback exists - call command callback */
144 if (cmd->callback != NULL) {
145 // original code wrap
146 if ((cmd->callback(context) != SCPI_RES_OK)) {
147 if (!context->cmd_error) {
148 SCPI_ErrorPush(context, SCPI_ERROR_EXECUTION_ERROR);
149 }
150 result = FALSE;
151 }
152 else {
153 if (context->cmd_error) {
154 result = FALSE;
155 }
156 }
157
158 }
159 /* set error if command callback did not read all parameters */
160 if (state->pos < (state->buffer + state->len) && !context->cmd_error) {
161 SCPI_ErrorPush(context, SCPI_ERROR_PARAMETER_NOT_ALLOWED);
162 result = FALSE;
163 }
164
165 return result;
166}

174static scpi_bool_t findCommandHeader(scpi_t* context, const char* header, int len) {
175 int32_t i;
176
177#ifdef __AMU_SCPI_USE_PROGMEM__
178 PGM_P pattern;
179
180 for (i = 0; (pattern = (PGM_P)pgm_read_word(&context->def_cmdlist[i].pattern)) != 0; i++) {
181 strncpy_P(context->param_list.cmd_pattern_s, pattern, SCPI_MAX_CMD_PATTERN_SIZE);
182 context->param_list.cmd_pattern_s[SCPI_MAX_CMD_PATTERN_SIZE] = '\0';
183
184 if (matchCommand(context->param_list.cmd_pattern_s, header, len, NULL, 0, 0, &context->query)) {
185 context->param_list.cmd_s.callback = (scpi_command_callback_t)pgm_read_word(&context->def_cmdlist[i].callback);
186 context->param_list.cmd_s.tag = (int32_t)pgm_read_dword(&context->def_cmdlist[i].tag);
187 return TRUE;
188 }
189 }
190
191 if (context->aux_cmdlist != NULL) {
192
193 for (i = 0; (pattern = (PGM_P)pgm_read_word(&context->aux_cmdlist[i].pattern)) != 0; i++) {
194 strncpy_P(context->param_list.cmd_pattern_s, pattern, SCPI_MAX_CMD_PATTERN_SIZE);
195 context->param_list.cmd_pattern_s[SCPI_MAX_CMD_PATTERN_SIZE] = '\0';
196
197 if (matchCommand(context->param_list.cmd_pattern_s, header, len, NULL, 0, 0, &context->query)) {
198 context->param_list.cmd_s.callback = (scpi_command_callback_t)pgm_read_word(&context->aux_cmdlist[i].callback);
199 context->param_list.cmd_s.tag = (int32_t)pgm_read_dword(&context->aux_cmdlist[i].tag);
200 return TRUE;
201 }
202 }
203 }
204#else
205 const scpi_command_t * cmd;
206
207 for (i = 0; context->def_cmdlist[i].pattern != NULL; i++) {
208 cmd = &context->def_cmdlist[i];
209 if (matchCommand(cmd->pattern, header, len, NULL, 0, 0, &context->query)) {
210 context->param_list.cmd = cmd;
211 return TRUE;
212 }
213 }
214
215 if (context->aux_cmdlist != NULL) {
216 for (i = 0; context->aux_cmdlist[i].pattern != NULL; i++) {
217 cmd = &context->aux_cmdlist[i];
218 if (matchCommand(cmd->pattern, header, len, NULL, 0, 0, &context->query)) {
219 context->param_list.cmd = cmd;
220 return TRUE;
221 }
222 }
223 }
224#endif
225
226 return FALSE;
227}

257scpi_bool_t SCPI_Parse(scpi_t * context, char * data, int len) {
258 scpi_bool_t result = TRUE;
259 scpi_parser_state_t * state;
260 int r;
261 scpi_token_t cmd_prev = {SCPI_TOKEN_UNKNOWN, NULL, 0};
262
263 if (context == NULL) {
264 return FALSE;
265 }
266
267 state = &context->parser_state;
268 context->output_count = 0;
269 context->query = false;
270 context->has_param = false;
271
272 while (1) {
273 r = scpiParser_detectProgramMessageUnit(state, data, len);
274
275 if (state->programHeader.type == SCPI_TOKEN_INVALID) {
276 SCPI_ErrorPush(context, SCPI_ERROR_INVALID_CHARACTER);
277 result = FALSE;
278 } else if (state->programHeader.len > 0) {
279
280 composeCompoundCommand(&cmd_prev, &state->programHeader);
281
282 if (findCommandHeader(context, state->programHeader.ptr, state->programHeader.len)) {
283
284 context->param_list.lex_state.buffer = state->programData.ptr;
285 context->param_list.lex_state.pos = context->param_list.lex_state.buffer;
286 context->param_list.lex_state.len = state->programData.len;
287 context->param_list.cmd_raw.data = state->programHeader.ptr;
288 context->param_list.cmd_raw.position = 0;
289 context->param_list.cmd_raw.length = state->programHeader.len;
290
291 result &= processCommand(context);
292
293 cmd_prev = state->programHeader;
294 } else {
295 /* place undefined header with error */
296 /* calculate length of errornouse header and trim \r\n */
297 size_t r2 = r;
298 while (r2 > 0 && (data[r2 - 1] == '\r' || data[r2 - 1] == '\n')) r2--;
299 SCPI_ErrorPushEx(context, SCPI_ERROR_UNDEFINED_HEADER, data, r2);
300 result = FALSE;
301 }
302 }
303
304 if (r < len) {
305 data += r;
306 len -= r;
307 } else {
308 break;
309 }
310
311 }
312
313 /* conditionaly write new line */
314 writeNewLine(context);
315
316 return result;
317}

392scpi_bool_t SCPI_Input(scpi_t * context, const char * data, int len) {
393 scpi_bool_t result = TRUE;
394 size_t totcmdlen = 0;
395 int cmdlen = 0;
396
397 if (len == 0) {
398 context->buffer.data[context->buffer.position] = 0;
399 result = SCPI_Parse(context, context->buffer.data, context->buffer.position);
400 context->buffer.position = 0;
401 } else {
402 int buffer_free;
403
404 buffer_free = context->buffer.length - context->buffer.position;
405 if (len > (buffer_free - 1)) {
406 /* Input buffer overrun - invalidate buffer */
407 context->buffer.position = 0;
408 context->buffer.data[context->buffer.position] = 0;
409 SCPI_ErrorPush(context, SCPI_ERROR_INPUT_BUFFER_OVERRUN);
410 return FALSE;
411 }
412 memcpy(&context->buffer.data[context->buffer.position], data, len);
413 context->buffer.position += len;
414 context->buffer.data[context->buffer.position] = 0;
415
416
417 while (1) {
418 cmdlen = scpiParser_detectProgramMessageUnit(&context->parser_state, context->buffer.data + totcmdlen, context->buffer.position - totcmdlen);
419 totcmdlen += cmdlen;
420
421 if (context->parser_state.termination == SCPI_MESSAGE_TERMINATION_NL) {
422 result = SCPI_Parse(context, context->buffer.data, totcmdlen);
423 memmove(context->buffer.data, context->buffer.data + totcmdlen, context->buffer.position - totcmdlen);
424 context->buffer.position -= totcmdlen;
425 totcmdlen = 0;
426 } else {
427 if (context->parser_state.programHeader.type == SCPI_TOKEN_UNKNOWN) break;
428 if (totcmdlen >= context->buffer.position) break;
429 }
430 }
431
432 }
433
434 return result;
435}

611 result += writeData(context, "\"", 1);
612 context->output_count++;

627size_t SCPI_ResultError(scpi_t * context, scpi_error_t * error) {
628 size_t result = 0;
629 size_t outputlimit = SCPI_STD_ERROR_DESC_MAX_STRING_LENGTH;
630 size_t step = 0;
631 const char * quote;
632
633 const char * data[SCPIDEFINE_DESCRIPTION_MAX_PARTS];
634 size_t len[SCPIDEFINE_DESCRIPTION_MAX_PARTS];
635 size_t i;
636
637 data[0] = SCPI_ErrorTranslate(error->error_code);
638 len[0] = strlen(data[0]);
639
640#if USE_DEVICE_DEPENDENT_ERROR_INFORMATION
641 data[1] = error->device_dependent_info;
642#if USE_MEMORY_ALLOCATION_FREE
643 len[1] = error->device_dependent_info ? strlen(data[1]) : 0;
644#else
645 SCPIDEFINE_get_parts(&context->error_info_heap, data[1], &len[1], &data[2], &len[2]);
646#endif
647#endif
648
649 result += SCPI_ResultInt32(context, error->error_code);
650 result += writeDelimiter(context);
651 result += writeData(context, "\"", 1);
652
653 for (i = 0; (i < SCPIDEFINE_DESCRIPTION_MAX_PARTS) && data[i] && outputlimit; i++) {
654 if (i == 1) {
655 result += writeSemicolon(context);
656 outputlimit -= 1;
657 }
658 if (len[i] > outputlimit) {
659 len[i] = outputlimit;
660 }
661
662 while ((quote = strnpbrk(data[i], len[i], "\""))) {
663 if ((step = quote - data[i] + 1) >= outputlimit) {
664 len[i] -= 1;
665 outputlimit -= 1;
666 break;
667 }
668 result += writeData(context, data[i], step);
669 result += writeData(context, "\"", 1);
670 len[i] -= step;
671 outputlimit -= step + 1;
672 data[i] = quote + 1;
673 if (len[i] > outputlimit) {
674 len[i] = outputlimit;
675 }
676 }
677
678 result += writeData(context, data[i], len[i]);
679 outputlimit -= len[i];
680 }
681 result += writeData(context, "\"", 1);
682
683 return result;
684}

772scpi_bool_t SCPI_Parameter(scpi_t * context, scpi_parameter_t * parameter, scpi_bool_t mandatory) {
773 lex_state_t * state;
774
775 if (!parameter) {
776 SCPI_ErrorPush(context, SCPI_ERROR_SYSTEM_ERROR);
777 return FALSE;
778 }
779
780 invalidateToken(parameter, NULL);
781
782 state = &context->param_list.lex_state;
783
784 if (state->pos >= (state->buffer + state->len)) {
785 if (mandatory) {
786 SCPI_ErrorPush(context, SCPI_ERROR_MISSING_PARAMETER);
787 } else {
788 parameter->type = SCPI_TOKEN_PROGRAM_MNEMONIC; /* TODO: select something different */
789 }
790 return FALSE;
791 }
792 if (context->input_count != 0) {
793 scpiLex_Comma(state, parameter);
794 if (parameter->type != SCPI_TOKEN_COMMA) {
795 invalidateToken(parameter, NULL);
796 SCPI_ErrorPush(context, SCPI_ERROR_INVALID_SEPARATOR);
797 return FALSE;
798 }
799 }
800
801 context->input_count++;
802
803 scpiParser_parseProgramData(&context->param_list.lex_state, parameter);
804
805 switch (parameter->type) {
806 case SCPI_TOKEN_HEXNUM:
807 case SCPI_TOKEN_OCTNUM:
808 case SCPI_TOKEN_BINNUM:
809 case SCPI_TOKEN_PROGRAM_MNEMONIC:
810 case SCPI_TOKEN_DECIMAL_NUMERIC_PROGRAM_DATA:
811 case SCPI_TOKEN_DECIMAL_NUMERIC_PROGRAM_DATA_WITH_SUFFIX:
812 case SCPI_TOKEN_ARBITRARY_BLOCK_PROGRAM_DATA:
813 case SCPI_TOKEN_SINGLE_QUOTE_PROGRAM_DATA:
814 case SCPI_TOKEN_DOUBLE_QUOTE_PROGRAM_DATA:
815 case SCPI_TOKEN_PROGRAM_EXPRESSION:
816 context->has_param = true;
817 return TRUE;
818 default:
819 invalidateToken(parameter, NULL);
820 SCPI_ErrorPush(context, SCPI_ERROR_INVALID_STRING_DATA);
821 return FALSE;
822 }
823}

1431int scpiParser_parseProgramData(lex_state_t * state, scpi_token_t * token) {
1432 scpi_token_t tmp;
1433 int result = 0;
1434 int wsLen;
1435 int suffixLen;
1436 int realLen = 0;
1437 realLen += scpiLex_WhiteSpace(state, &tmp);
1438
1439 if (result == 0) result = scpiLex_NondecimalNumericData(state, token);
1440 if (result == 0) result = scpiLex_CharacterProgramData(state, token);
1441 if (result == 0) {
1442 result = scpiLex_DecimalNumericProgramData(state, token);
1443 if (result != 0) {
1444 wsLen = scpiLex_WhiteSpace(state, &tmp);
1445 suffixLen = scpiLex_SuffixProgramData(state, &tmp);
1446 if (suffixLen > 0) {
1447 token->len += wsLen + suffixLen;
1448 token->type = SCPI_TOKEN_DECIMAL_NUMERIC_PROGRAM_DATA_WITH_SUFFIX;
1449 result = token->len;
1450 }
1451 }
1452 }
1453
1454 if (result == 0) result = scpiLex_StringProgramData(state, token);
1455 if (result == 0) result = scpiLex_ArbitraryBlockProgramData(state, token);
1456 if (result == 0) result = scpiLex_ProgramExpression(state, token);
1457
1458 realLen += scpiLex_WhiteSpace(state, &tmp);
1459
1460 return result + realLen;
1461}

1470int scpiParser_parseAllProgramData(lex_state_t * state, scpi_token_t * token, int * numberOfParameters) {
1471
1472 int result;
1473 scpi_token_t tmp;
1474 int paramCount = 0;
1475
1476 token->len = -1;
1477 token->type = SCPI_TOKEN_ALL_PROGRAM_DATA;
1478 token->ptr = state->pos;
1479
1480
1481 for (result = 1; result != 0; result = scpiLex_Comma(state, &tmp)) {
1482 token->len += result;
1483
1484 if (result == 0) {
1485 token->type = SCPI_TOKEN_UNKNOWN;
1486 token->len = 0;
1487 paramCount = -1;
1488 break;
1489 }
1490
1491 result = scpiParser_parseProgramData(state, &tmp);
1492 if (tmp.type != SCPI_TOKEN_UNKNOWN) {
1493 token->len += result;
1494 } else {
1495 token->type = SCPI_TOKEN_UNKNOWN;
1496 token->len = 0;
1497 paramCount = -1;
1498 break;
1499 }
1500 paramCount++;
1501 }
1502
1503 if (token->len == -1) {
1504 token->len = 0;
1505 }
1506
1507 if (numberOfParameters != NULL) {
1508 *numberOfParameters = paramCount;
1509 }
1510 return token->len;
1511}

1520int scpiParser_detectProgramMessageUnit(scpi_parser_state_t * state, char * buffer, int len) {
1521 lex_state_t lex_state;
1522 scpi_token_t tmp;
1523 int result = 0;
1524
1525 lex_state.buffer = lex_state.pos = buffer;
1526 lex_state.len = len;
1527 state->numberOfParameters = 0;
1528
1529 /* ignore whitespace at the begginig */
1530 scpiLex_WhiteSpace(&lex_state, &tmp);
1531
1532 if (scpiLex_ProgramHeader(&lex_state, &state->programHeader) >= 0) {
1533 if (scpiLex_WhiteSpace(&lex_state, &tmp) > 0) {
1534 scpiParser_parseAllProgramData(&lex_state, &state->programData, &state->numberOfParameters);
1535 } else {
1536 invalidateToken(&state->programData, lex_state.pos);
1537 }
1538 } else {
1539 invalidateToken(&state->programHeader, lex_state.buffer);
1540 invalidateToken(&state->programData, lex_state.buffer);
1541 }
1542
1543 if (result == 0) result = scpiLex_NewLine(&lex_state, &tmp);
1544 if (result == 0) result = scpiLex_Semicolon(&lex_state, &tmp);
1545
1546 if (!scpiLex_IsEos(&lex_state) && (result == 0)) {
1547 lex_state.pos++;
1548
1549 state->programHeader.len = 1;
1550 state->programHeader.type = SCPI_TOKEN_INVALID;
1551
1552 invalidateToken(&state->programData, lex_state.buffer);
1553 }
1554
1555 if (SCPI_TOKEN_SEMICOLON == tmp.type) {
1556 state->termination = SCPI_MESSAGE_TERMINATION_SEMICOLON;
1557 } else if (SCPI_TOKEN_NL == tmp.type) {
1558 state->termination = SCPI_MESSAGE_TERMINATION_NL;
1559 } else {
1560 state->termination = SCPI_MESSAGE_TERMINATION_NONE;
1561 }
1562
1563 return lex_state.pos - lex_state.buffer;
1564}
