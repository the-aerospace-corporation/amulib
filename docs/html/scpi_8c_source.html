146// If commmand number exists, it's placed in the first byte of the transfer register
147// NO parameters, otherwise error is thrown.
148scpi_result_t scpi_cmd_execute(scpi_t* context) {
149
150 int32_t* commandNumber = (int32_t*)scpi_amu_dev->transfer_reg;
151
152 SCPI_CommandNumbers(context, commandNumber, 1, -1);
153
154 _scpi_get_channelList(context);
155
156 for (uint8_t* device = scpi_channel_list; *device != AMU_DEVICE_END_LIST; device++) {
157 if (*commandNumber == -1)
158 _amu_route_command(*device, SCPI_CmdTag(context), 0, false);
159 else
160 _amu_route_command(*device, SCPI_CmdTag(context), 1, false);
161 }
162 return SCPI_RES_OK;

165scpi_result_t _scpi_read_ptr(scpi_t* context) {
166
167 uint8_t numPoints;
168
169 _scpi_get_channelList(context);
170
171 for (uint8_t* device = scpi_channel_list; *device != AMU_DEVICE_END_LIST; device++) {
172
173 _amu_route_command(*device, AMU_REG_SWEEP_CONFIG_NUM_POINTS, 1, true);
174 numPoints = scpi_amu_dev->transfer_reg[0];
175
176 switch ((AMU_REG_DATA_PTR_t)SCPI_CmdTag(context)) {
177 case AMU_REG_DATA_PTR_TIMESTAMP:
178 _amu_route_command(*device, SCPI_CmdTag(context), numPoints * sizeof(uint32_t), true);
179 SCPI_ResultArrayUInt32(context, (uint32_t*)&scpi_amu_dev->transfer_reg[0], numPoints, SCPI_FORMAT_ASCII);
180 break;
181 case AMU_REG_DATA_PTR_VOLTAGE:
182 case AMU_REG_DATA_PTR_CURRENT:
183 case AMU_REG_DATA_PTR_SS_YAW:
184 case AMU_REG_DATA_PTR_SS_PITCH:
185 _amu_route_command(*device, SCPI_CmdTag(context), numPoints * sizeof(float), true);
186 SCPI_ResultArrayFloat(context, (float*)&scpi_amu_dev->transfer_reg[0], numPoints, SCPI_FORMAT_ASCII);
187 break;
188 case AMU_REG_DATA_PTR_SWEEP_CONFIG:
189 _amu_route_command(*device, SCPI_CmdTag(context), sizeof(ivsweep_config_t), true);
190 SCPI_ResultArrayUInt8(context, (uint8_t*)&scpi_amu_dev->transfer_reg[0], 8, SCPI_FORMAT_ASCII);
191 SCPI_ResultArrayFloat(context, (float*)&scpi_amu_dev->transfer_reg[8], 2, SCPI_FORMAT_ASCII);
192 break;
193 case AMU_REG_DATA_PTR_SWEEP_META:
194 _amu_route_command(*device, SCPI_CmdTag(context), sizeof(ivsweep_meta_t), true);
195 SCPI_ResultArrayFloat(context, (float*)&scpi_amu_dev->transfer_reg[0], 10, SCPI_FORMAT_ASCII);
196 SCPI_ResultArrayUInt32(context, (uint32_t*)&scpi_amu_dev->transfer_reg[40], 2, SCPI_FORMAT_ASCII);
197 break;
198 case AMU_REG_DATA_PTR_SUNSENSOR:
199 _amu_route_command(*device, SCPI_CmdTag(context), sizeof(ss_angle_t), true);
200 SCPI_ResultArrayFloat(context, (float*)&scpi_amu_dev->transfer_reg[0], 2, SCPI_FORMAT_ASCII);
201 break;
202 case AMU_REG_DATA_PTR_PRESSURE:
203 _amu_route_command(*device, SCPI_CmdTag(context), sizeof(press_data_t), true);
204 SCPI_ResultArrayFloat(context, (float*)&scpi_amu_dev->transfer_reg[0], 4, SCPI_FORMAT_ASCII);
205 break;
206 default: break;
207 }
208
209 }
210
211 return SCPI_RES_OK;

282scpi_result_t _scpi_cmd_measure_channel(scpi_t* context) {
283
284 int32_t channel = -1;
285
286 memset((void*)scpi_amu_dev->transfer_reg, 0, 4);
287
288 SCPI_CommandNumbers(context, &channel, 1, -1);
289
290 if (channel == -1)
291 channel = SCPI_CmdTag(context) & 0x0F; // So instead, use the tag
292 else
293 scpi_amu_dev->transfer_reg[0] = channel;
294
295 if (channel > 15)
296 return SCPI_RES_ERR;
297
298 if (!context->query) {
299 return SCPI_RES_ERR;
300 }
301
302 _scpi_get_channelList(context);
303
304 for (uint8_t* device = scpi_channel_list; *device != AMU_DEVICE_END_LIST; device++) {
305
306 if (SCPI_CmdTag(context) >= CMD_I2C_USB)
307 _amu_route_command(*device, (SCPI_CmdTag(context) | CMD_READ), 4, true);
308 else
309 _amu_route_command(*device, SCPI_CmdTag(context), 4, true);
310
311
312 if (strstr(context->param_list.cmd_raw.data, "RAW"))
313 SCPI_ResultUInt32Base(context, scpi_amu_dev->amu_regs->adc_raw.channel[channel], 16);
314 else
315 SCPI_ResultFloat(context, transfer_read_float());
316 }
317
318 return SCPI_RES_OK;
319}

321scpi_result_t _scpi_cmd_measure_active_ch(scpi_t* context) {
322
323 _scpi_get_channelList(context); // Get the list of channels to act on
324
325 uint16_t* activeChannels = (uint16_t*)scpi_amu_dev->transfer_reg;
326 uint8_t numChannels = 0;
327
328 for (uint8_t* device = scpi_channel_list; *device != AMU_DEVICE_END_LIST; device++) {
329
330 _amu_route_command(*device, AMU_REG_SYSTEM_ADC_ACTIVE_CHANNELS, sizeof(uint16_t), true);
331
332 for (uint16_t i = 0; i < 16; i++) {
333 if (*activeChannels & (1 << i))
334 numChannels++;
335 }
336
337 _amu_route_command(*device, SCPI_CmdTag(context), numChannels * 4, false);
338
339 if (numChannels > 0) {
340 SCPI_ResultArrayFloat(context, (float*)scpi_amu_dev->transfer_reg, numChannels, SCPI_FORMAT_ASCII);
341 }
342 }
343
344 return SCPI_RES_OK;
345}

620int16_t _scpi_get_channelList(scpi_t* context) {
621
622 scpi_parameter_t channel_list_param;
623 uint8_t scpi_list_iterator = 0;
624
625 //Checks for any parameter
626 if (SCPI_Parameter(context, &channel_list_param, FALSE)) {
627
628 scpi_bool_t is_range;
629 size_t dimensions;
630
631 /* the next statement is valid usage and it gets only real number of dimensions for the first item (index 0) */
632 if (!SCPI_ExprChannelListEntry(context, &channel_list_param, 0, &is_range, NULL, NULL, 0, &dimensions)) {
633
634 size_t param_idx = 0; /* index for channel list */
635 int8_t direction = 1; /* direction of counter for rows, +/-1 */
636 int32_t address_start = 0;
637 int32_t address_end = 0;
638
639
640 while (SCPI_EXPR_OK == SCPI_ExprChannelListEntry(context, &channel_list_param, param_idx, &is_range, &address_start, &address_end, 1, &dimensions)) {
641
642 if ((dimensions != 1) | (address_start > 63)) {
643 scpi_channel_list[scpi_list_iterator] = AMU_DEVICE_END_LIST;
644 return FALSE;
645 }
646
647 if (is_range) {
648
649 if (address_end > 63) {
650 scpi_channel_list[scpi_list_iterator] = AMU_DEVICE_END_LIST;
651 return FALSE;
652 }
653
654 for ((address_start > address_end) ? (direction = -1) : (direction = 1); address_start != address_end; address_start += direction)
655 scpi_channel_list[scpi_list_iterator++] = address_start;
656
657 scpi_channel_list[scpi_list_iterator++] = address_start;
658
659 }
660 else {
661 scpi_channel_list[scpi_list_iterator++] = address_start;
662 }
663
664 /* increase index */
665 param_idx++;
666 }
667 }
668 }
669 else { //No parameter list
670 scpi_channel_list[scpi_list_iterator++] = 0;
671 }
672
673 scpi_channel_list[scpi_list_iterator] = AMU_DEVICE_END_LIST;
674
675 return TRUE;
676}
